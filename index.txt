// git add some changes to stage
git add -A
git add "*.html"
git add "some.*"


// show git status
git status


// commit stage changes
git commit -m 'some messages'

// git show change details
git diff HEAD
git diff --staged


// reset stage changes
git reset -A
git reset "*.html"
git reset "some.*"


// git recover file changes
git checkout -- index.html
